% Análisis de Datos Oceanógraficos
% Claudio Iturra, 2015
% DOCE, Universidad de Concepcion.

% Para la actividad 1A de hoy deben extraer un polígono de la matriz de datos 
% de velocidad (u,v), y crear una imagen nueva con los datos extraidos. 
% Recordar graficar la linea de costa del área de estudio.

% Ejemplo
% Cuadrado

clear all;,close all;

xq = rand(1,100);                                           % Coordenadas X
yq = rand(1,100);                                           % Coordenadas Y
xv = [0.3, 0.3, 0.7, 0.7, 0.3];                             % Coordenadas del Poligono en eje X
yv = [0.3, 0.7, 0.7, 0.3, 0.3];                             % Coordenadas del Poligono en eje Y
[in,on] = inpolygon(xq,yq, xv,yv);                          % Matrix
inon = in | on;                                             % Combine ‘in’ And ‘on’
idx = find(inon(:));                                        % Indices de puntos dentro del poligono
xcoord = xq(idx);                                           % Coordenadas x de puntos dentro del poligono
ycoord = yq(idx);                                           % Coordenadas y de puntos dentro del poligono
figure(1)
plot(xq, yq, '+')                                           % Graficar todos los datos
hold on
plot(xv, yv, '-r')                                          % Plotear Poligono
plot(xcoord, ycoord, '+')                                   % Overplot  inon’ Points
hold off
xlabel('X');,ylabel('Y')
