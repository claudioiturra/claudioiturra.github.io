% get wind stress as tau = rho_air * drag_coeff(|u_10|) * |u_10| * u_10
% (That's inefficient but much easier to read.)
taux = ...
    rho_air ...
    * drag_coefficient(uv_mag(u10_model, v10_model)) ...
    .* uv_mag(u10_model, v10_model) ...
    .* u10_model;
tauy = ...
    rho_air ...
    * drag_coefficient(uv_mag(u10_model, v10_model)) ...
    .* uv_mag(u10_model, v10_model) ...
    .* v10_model;

% integrate
for l = 3 : length(t)

    % update u and v
    u(l) = u(l-2) ...
        + 2 * dt * f * v(l-1) ...
        - 2 * dt * epsilon * u(l-2) ...
        + 2 * dt * taux(l-2) / rho0 / H;

    v(l) = v(l-2) ...
        - 2 * dt * f * u(l-1) ...
        - 2 * dt * epsilon * v(l-2) ...
        + 2 * dt * tauy(l-2) / rho0 / H;

    if mod(l, round(num_time_steps/10)) == 0
    %fprintf('... l = %05d done after %f seconds\n', l, toc);
  end
end
